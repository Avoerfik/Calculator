15void ExpressionAnalyzer::analyzeOperators(const std::string& expression)
16{
17 // Анализ паттернов использования операторов
18 const std::string operators = "+-*/^";
19
20 // 1. Проверка на последовательные операторы (кроме унарных +- в начале)
21 bool hasConsecutiveOps = false;
22 for (size_t i = 1; i < expression.length(); ++i) {
23 if (operators.find(expression[i]) != std::string::npos &&
24 operators.find(expression[i - 1]) != std::string::npos) {
25 // Исключаем случаи унарных операторов: "+5", "-3", "(-2)", "+(5)"
26 if (!(expression[i] == '+' || expression[i] == '-') ||
27 (i > 0 && expression[i - 1] != '(' && expression[i - 1] != ' ')) {
28 hasConsecutiveOps = true;
29 break;
30 }
31 }
32 }
33
34 // 2. Проверка на операторы в начале/конце (кроме унарных +-)
35 bool startsWithBinaryOp = false;
36 bool endsWithOp = false;
37
38 if (!expression.empty()) {
39 char firstChar = expression[0];
40 if (operators.find(firstChar) != std::string::npos &&
41 firstChar != '+' && firstChar != '-') {
42 startsWithBinaryOp = true;
43 }
44
45 char lastChar = expression[expression.length() - 1];
46 if (operators.find(lastChar) != std::string::npos) {
47 endsWithOp = true;
48 }
49 }
50
51 // 3. Анализ баланса операторов и операндов
52 int operatorCount = 0;
53 int numberCount = 0;
54 const std::string numbers = "0123456789";
55
56 for (char c : expression) {
57 if (operators.find(c) != std::string::npos) {
58 operatorCount++;
59 }
60 if (numbers.find(c) != std::string::npos) {
61 numberCount++;
62 }
63 }
64
65 // 4. Проверка на использование оператора деления с нулем
66 bool hasDivisionByZeroPattern = false;
67 size_t divPos = expression.find('/');
68 if (divPos != std::string::npos && divPos + 1 < expression.length()) {
69 // Ищем следующий символ после '/'
70 size_t nextPos = divPos + 1;
71 // Пропускаем пробелы и унарные +-
72 while (nextPos < expression.length() &&
73 (expression[nextPos] == ' ' || expression[nextPos] == '+' || expression[nextPos] == '-')) {
74 nextPos++;
75 }
76
77 if (nextPos < expression.length()) {
78 // Проверяем, не идет ли сразу ноль
79 if (expression[nextPos] == '0') {
80 // Но исключаем случаи типа "0.5"
81 if (nextPos + 1 >= expression.length() || expression[nextPos + 1] != '.') {
82 hasDivisionByZeroPattern = true;
83 }
84 }
85 }
86 }
87
88 // 5. Анализ приоритетов операторов
89 bool hasHighPriorityOps = expression.find('^') != std::string::npos;
90 bool hasMediumPriorityOps = expression.find('*') != std::string::npos ||
91 expression.find('/') != std::string::npos;
92 bool hasLowPriorityOps = expression.find('+') != std::string::npos ||
93 expression.find('-') != std::string::npos;
94
95 // 6. Сохраняем результаты анализа (можно добавить в структуру ExpressionStats)
96 // Пока просто выводим в консоль для отладки
97 std::cout << "\n=== Operator Analysis ===\n";
98 std::cout << "Expression: " << expression << "\n";
99 std::cout << "Consecutive operators: " << (hasConsecutiveOps ? "YES" : "NO") << "\n";
100 std::cout << "Starts with binary operator: " << (startsWithBinaryOp ? "YES" : "NO") << "\n";
101 std::cout << "Ends with operator: " << (endsWithOp ? "YES" : "NO") << "\n";
102 std::cout << "Operator count: " << operatorCount << "\n";
103 std::cout << "Number count: " << numberCount << "\n";
104 std::cout << "Operand/Operator ratio: " << (operatorCount > 0 ? static_cast<double>(numberCount) / operatorCount : 0) << "\n";
105 std::cout << "Potential division by zero: " << (hasDivisionByZeroPattern ? "WARNING" : "NO") << "\n";
106 std::cout << "High priority ops (^): " << (hasHighPriorityOps ? "YES" : "NO") << "\n";
107 std::cout << "Medium priority ops (*,/): " << (hasMediumPriorityOps ? "YES" : "NO") << "\n";
108 std::cout << "Low priority ops (+,-): " << (hasLowPriorityOps ? "YES" : "NO") << "\n";
109
110 // 7. Проверка на сложные операторные конструкции
111 bool hasComplexOperatorPatterns = hasConsecutiveOps || startsWithBinaryOp || endsWithOp || hasDivisionByZeroPattern;
112 if (hasComplexOperatorPatterns) {
113 std::cout << "WARNING: Expression has complex operator patterns that might cause errors!\n";
114 }
115}

117void ExpressionAnalyzer::analyzeExpression(const std::string& expression, double calcTime)
118{
119 ExpressionStats stats;
120 stats.expression = expression;

122
123 // Проверяем, что complexityCalc установлен (агрегация)
