41{
42 bool isPassed;
43 std::cout << "№" << testNumber << ":\n";
44 try
45 {
46 std::stack<std::string> expressionStack = bracketAnalyze(expression);
47 std::string result = calculateExpression(expressionStack);
48 std::cout << "Test: " << testName << " - FAILED (should have thrown exception)\n";
49 std::cout << "Result was: " << result << '\n';
50 isPassed = false;
51 }
52 catch (const std::runtime_error& e)
53 {
54 if (std::string(e.what()).find(exceptionName) != std::string::npos)
55 {
56 std::cout << "Test: " << testName << " - PASSED (correctly caught format error)\n";
57 isPassed = true;
58 }
59 else
60 {
61 std::cout << "Test: " << testName << " - FAILED (wrong exception: " << e.what() << ")\n";
62 isPassed = false;
63 }
64 }
65 catch (const std::exception& e)
66 {
67 std::cout << "Test: " << testName << " - FAILED (unexpected exception: " << e.what() << ")\n";
68 isPassed = false;
69 }
70 std::cout << "----------------------\n";
71
72 return isPassed;
